نبی کا قربانی کے جانوروں کو زندہ زخمی کے انکا مثلہ کرنا اور پھر جوتیوں کے ہار پہنانا

 اگر آپ کو شک ہے کہ محمد نے قربانی کے نام پر عرب کی جاہل روایات کا اتباع نہیں کیا ہے تو محمد کا عمل دیکھئے جہاں وہ قربانی کے نام پر زندہ اونٹ/اونٹی کے کوہان کو چیر کر مثلہ کر رہے ہیں جسے اشعار کہا جاتا ہے۔ اور پھر محمد صلعم قربانی کے جانور کے گلے میں جوتوں کا ہار ڈال رہے ہیں جسے قلادہ کہا جاتا ہے۔

صحیح مسلم ، ذبح کا بیان (لنک):

آپ صلی اللہ علیہ وآلہ وسلم نے اپنی اونٹنی کو منگوایا اور اس کے کوہان میں دائیں طرف شعار (زندہ جانور کو زخمی کر کے خون نکالنا) کیا جس سے خون بہا اور اس کے گلے میں دو جوتیوں کا ہار ڈالا۔

کیا جوتیوں کا ہار پہنانا جہالت نہیں ؟

اور زندہ جانور کو زخمی کر کے مثلہ کرنا کونسی عقلمندی کی بات ہے؟

اللہ کو بھلا جانوروں کو جوتیوں کا ہار پہنانے اور زندہ جانروں کو زخم دینے کی کیا ضرورت ہے؟

امام ترمذیؒ اس حدیث کے تحت لکھتے ہیں:

 والعمل علی ھذا عند أھل العلم من أصحاب النّبیّ صلّی اللہ علیہ وسلّم و غیرھم، یرون الإشعار، وھو قول الثّوریّ والشّافعیّ و أحمد وإسحاق۔

ترجمہ:
اسی پر نبی اکرم کے صحابہ اور دوسرے اہل علم کا عمل ہے، وہ اشعار کو جائز سمجھتے ہیں۔ امام سفیان ثوری، امام شافعی، امام احمد بن حنبل اور امام اسحاق بن راہویہؒ کا بھی یہی مذہب ہے۔
(سنن الترمذی، تحت حدیث: ۹۰۶)

صحابہ بھی اس رسم پر عمل پیرا تھے

موطا امام مالک، کتاب الحج (لنک):
نافع سے روایت ہے کہ عبداللہ بن عمر جب اپنی ہدی کے کوہان میں زخم لگاتے اشعار کے لئے کہتے اللہ کے نام سے اللہ بڑا ہے۔

موطا امام مالک، کتاب الحج (لنک):

عبداللہ بن عمر جب ہدی (قربانی کا جانور) لے جاتے مدینہ سے تو تقلید کرتے اس کی اور اشعار کرتے اس کا ذوالحلیفہ میں، مگر تقلید اشعار سے پہلے کرتے لیکن دونوں ایک ہی مقام میں کرتے اس طرح پر کہ ہدی کا منہ قبلہ کی طرف کر کے پہلے اس کے گلے میں دو جوتیاں لٹکا دیتے پھر اشعار (زخم لگاتے) کرتے بائیں طرف سے ۔

جانوروں کو جوتیوں کے ہار پہنانا اور اشعار وغیرہ ایسی جاہلانہ رسم ہے کہ آج کے مسلمان نبی صلعم کی سنت کو پیارا رکھنے کے باوجود آج یہ حرکت نہیں کرتے۔

اور یاد رکھئے کہ نبوت سے قبل محمد بتوں کے آگے جانور قربان کرتے، پھر یہ حرام گوشت کھاتے، اور بتوں کی پرستش کرتے تھے۔

صحیح بخاری، ذبیحہ و شکار کا بیان (لنک):

عبداللہ  آپ صلی اللہ علیہ وسلم سے روایت کرتے ہیں کہ آپ نے زید بن عمرو بن نفیل سے مقام اسفل بلدح میں ملاقات کی اور یہ واقعہ نبی صلی اللہ علیہ وسلم پر وحی نازل ہونے سے پہلے کا ہے، آپ صلی اللہ علیہ وسلم نے ان کے سامنے دسترخوان پیش کیا، جس پر گوشت تھا، انہوں نے اس کے کھانے سے انکار کیا، پھر فرمایا میں اس سے نہیں کھاتا ہوں، جس کو تم نے اپنے بتوں پر ذبح کرتے ہو اور میں صرف اسی کو کھاتا ہوں، جس پر اللہ کا نام لیا گیا ہو، یعنی بسم اللہ پڑھی گئی ہو۔

  

 

ابو حنیفہ کا شعار کو مثلہ کہنا، اور اس سے منع کرنا

زندہ اونٹوں کے کوہانوں کو کاٹ کر خون نکالنا ایسی خونخواری تھی کہ ابو حنیفہ نے اسے مثلہ قرار دیتے ہوئے اس سے منع کر دیا۔ مگر اس پر دوسرے مسلمان ابو حنیفہ کو برا بھلا کہنا شروع ہو گئے۔ 

مگر ان کے ابو حنیفہ کو برا بھلا کہنے کے باوجود مسلمان خود اس زندہ جانوروں کے مثلے سے دور ہوتے چلے گئے اور انہوں نے محمد کی یہ سنت بھی ترک کر دی جہاں جانوروں کو جوتوں کا ہار پہنایا جاتا تھا۔ وجہ یہ ہے کہ جوتوں کا ہار پہنانا اتنی بڑی جہالت تھی کہ لاکھ سنت اور شریعت کا نام دینے کے انسان کا شعور اسے ماننے کے لیے راضی نہیں ہوتا۔ 

لنک:

۱۔ حافظ نوویؒ (۶۴۱۔۶۷۶ھ) لکھتے ہیں:

 و قال أبو حنیفۃ: الإشعار بدعۃ، لأنّہ یخالف الٔاحادیث الصحیۃ المشھور فی الإشعار، وأمّا قولہ: إنّہ مثلۃ، فلیس کذلک، بل ھذا کالفصد والحجامۃ والکّی والوسم۔

 ‘‘ امام ابوحنیفہ نے کہا ہے کہ اشعار بدعت ہے، کیونکہ یہ مثلہ ہے۔ ان کا یہ قول اشعار کے بارے میں بہت سی صحیح اور مشہور احادیث کے خلاف ہے۔ رہا ان کا اشعار کو مثلہ کہنا تو یہ درست نہیں، کیونکہ اشعار ایسے ہی ہے، جیسے فصد، سنگی، داغ دینا اور نشان لگانا ہوتاہے۔’’

(شرح صحیح مسلم للنووی: ۴۰۷/۱)

۲۔ امام وکیع بن جراحؒ (م ۱۷۹ھ) فرماتے ہیں:

 لا تنظروا إلی قول أھل الرّأی فی ھذا، فإنّ الإشعار سنّۃ، وقولھم بدعۃ۔

 ‘‘ تم اس بارے میں اہل الرائے (ابوحنیفہ اور ان کے اصحاب) کے قول کو نہ دیکھو۔ اشعار سنت ہے، جبکہ (اس کو بدعت کہنے پر مبنی) ان کا قول خود بدعت ہے۔’’

(سنن الترمذی، تحت حدیث: ۹۰۶،وسندہٗ صحیحٌ)

ابوالسائب بن جنادہ کہتے ہیں:

 کنا عند و کیع، فقال لرجل عندہ ممن ینظر فی الرأی: أشعر رسول اللہ صلی اللہ علیہ وسلم، ویقول أبو حنیفۃ: ھو مثلۃ، قال الرجل: فإنہ قد روی عن إبراھیم النخعی أنہ قال الإشعار مثلۃ، قال فرأیت وکیعا غضب غضبا شدیدا، وقال: أقول لک قال رسول اللہ صلی اللہ علیہ وسلم و تقول: قال إبراھیم، ما أحقک بأن تجس ثم لا تخرج حتی تنزع عن قولک ھذا۔

 ترجمہ:
‘‘ ہم امام وکیعؒ کے پاس تھے۔ انہوں نے اپنے پاس بیھٹے ہوئے ایک آدمی، جو کہ رائے میں دلچسپی رکھتا تھا، سے فرمایا، اللہ کے رسول نے اشعار کیا ہے، امام ابو حنیفہ کہتے ہیں کہ یہ مثلہ ہے! آدمی کہنے لگا، ابراہیم نخعیؒ سے مروی ہے کہ انہوں نے اشعار کو مثلہ کہا ہے۔ راوی کہتے ہیں کہ میں نے دیکھا کہ امام وکیعؒ سخت غصہ میں آ گئے اور فرمانے لگے، میں تجھے رسول اللہ کی حدیث سناتا ہوں اور تو کہتا ہے کہ ابراہیم نخعی اس طرح کہتے ہیں۔ میں تجھے اس قابل سمجھتا ہوں کہ تجھے قید کر لیا جائے اور اس وقت تک نہ چھوڑا جائے، جب تک تو اپنے اس قول سے باز نہ آ جائے۔’’

(سنن الترمذی، تحت حدیث: ۹۰۶، وسندہٗ صحیحٌ) ۔۔۔

۳۔ امام ابن خزیمہؒ (م ۳۱۱ھ) حدیث ابن عباس پر یوں باب قائم کرتے ہیں:

 باب إشعار البُدن فی شقّ السّنام الأیمن، وسلت الدّم عنھا، ضدّقول من زعم أنّ إشعار البُدن مثلۃ، فسمّی سنّۃ النّبیّ صلّی اللہ علیہ وسلّم مثلۃ بجھلہ۔

 ترجمہ:

‘‘ قربانی کے اونٹوں کی کوہان کی دائیں جانب اشعار کرنے اور خون کو لتھڑنے کا بیان، اس شخص کے ردّ میں جو یہ دعوی کرتا ہے کہ اونٹوں کو اشعار کرنا مثلہ ہے، اس نے اپنی جہالت کی وجہ سے نبی اکرم کی سنت کا نام مثلہ رکھ دیا ہے۔’’

(صحیح ابن خزیمۃ: ۱۵۳/۴، ح: ۲۵۷۵)

۴۔ امام ابن عبدالبرؒ (م ۴۶۳ھ) لکھتے ہیں:

 و ھذا الحکم لا دلیل علیہ إلّا التّوھّم والظّنّ، ولا تترک السّنن بالظّنون۔

 ترجمہ:

‘‘ (امام ابو حنیفہ کے) اس فیصلے پر کوئی دلیل نہیں، سوائے توہم پرستی اور ظن وتخمین کے، جبکہ سنتوں کو ظن و تخمین کی وجہ سے نہیں چھوڑا جا سکتا۔’’

(الاستذکار لابن عبدالبر: ۲۶۴/۴)

۵۔ علامہ ابن حزمؒ (م ۴۵۶ھ) اس بارے میں لکھتے ہیں:

فقال أبو حنیفۃ: أکرہ الإشعار، وھومثلۃ، قال علیّ: ھذا طامّۃ من طوام العالم أن یکون مثلۃ شیء فعلہ النبی صلی اللہ علیہ وسلم، أف لکل عقل یتعقب حکم رسول اللہ صلی اللہ علیہ وسلم ویلزمہ أن تکون الحجامۃ، وفتح العرق مثلہ، فیمنع من ذلک، وأن یکون القصاص من قطع الأنف، و قلع الأسنان، وجدع الاذنین مثلۃ، و أن یکون قطع السارق و المحارب مثلۃ، والرجم للزانی المحصن مثلۃ، والصلب للمحارب مثلۃ، إنما المثلۃ فعل من بلغ نفسہ مبلغ انتقاد فعل رسول اللہ صلی اللہ علیہ وسلم، فھذا ھو الذی مثل بنفسہ، والإشعار کان فی حجۃ الوداع والنھی عن المثلۃ کان قبل قیام ذلک بأعوام، فصح أنہ لیس مثلۃ، وھذہ قولۃ لا یعلم لابی حنیفۃ فیھا متقدم من السلف، ولا موافق من فقھاء أھل عصرہ إلا من ابتلاہ اللہ بتقلیدہ، و نعوذ باللہ من البلاء۔

 ترجمہ:

‘‘ امام ابو حنیفہ نے کہا ہے کہ میں اشعار کو مکروہ سمجھتا ہوں، یہ تو مثلہ ہے، لیکن یہ کسی عالم کی ہفوات میں سے ہے کہ جس کام کو رسول اللہ نے کیا ہے، اسے وہ مثلہ قرار دے۔ ہر اس شخص پر افسوس ہے، جو رسول اللہ کے فیصلے پر گرفت کرتا ہے۔ ایسی عقل پر یہ لازم آتا ہے کہ اس کے نزدیک سنگی لگوانا، فصد کھولنا وغیرہ بھی مثلہ ہو اور وہ اس سے بھی رک جائے، نیز اس کے نزدیک ناک کاٹنے، دانت اکھیڑنے، کان کاٹنے وغیرہ کا قصاص لینا بھی مثلہ ہو اور چوری اور فسادی آدمی کا ہاتھ کاٹنا بھی مثلہ ہو، شادی شدہ زانی کو سنگسار کرنا بھی مثلہ ہو، زمین میں فساد کرنے والے کو سولی دینا بھی مثلہ ہو۔ دراصل مثلہ تو اس نے کیا ہے، جس نے اپنے آپ کو رسول اللہ کے فعل مبارک پر تنقید تک پہنچا دیا ہے، یہ وہ شخص ہے، جس نے اپنے نفس کا مثلہ کیا ہے۔ حالانکہ اشعار حجۃ الوداع میں کیا گیا تھا اور مثلہ سے ممانعت اس سے کئی سال پہلے ہو چکی تھی۔ ثابت ہو کہ یہ مثلہ نہیں۔

یہ امام ابو حنیفہ کا ایسا قول ہے، جس میں ان کا کوئی سلف نہیں، نہ ہی ان کے ہم زمانہ فقہائے کرام میں سے کسی نے ان کی موافقت کی ہے، سوائے ان لوگوں کے جن کو اللہ تعالی نے ان کی تقلید کی آزمائش میں ڈال دیا ہے۔ ہم فتنہ (تقلید) سے اللہ تعالی کی پناہ میں آتے ہیں۔’’

(المحلی لابن حزم: ۱۱۲-۱۱۱/۷)

 

 

Prophet Muhammad injuring the live animals and then putting shoes around their necks

Prohet Muhammad used to put the necklace of Shoes around neck of Animals before sacrifice
Sahih Muslim, Book of Hajj (link):
Allah's Messenger called for his she-camel and did the "Asha'ar" i.e. to cut the right side of its bump, and removing the blood from it, and tied two shoes round its neck as necklace.
Is it not pure Ignorance?
Why Allah is in need of cutting the live animal for blood, and why Allah in need of these animals wearing necklace of Shoes?
For sure, these traditions of "cutting of live animals for blood" and "necklace of shoes" also came from the Kuffar Culture, same as the sacrifice itself. Islamic religion adopted some practices from the Jews/Christians, and some practices from the Kuffar culture, especially those practices which were related to Kaba.
The companions were also practicing this tradition:
Muwatta Imam Malik (link):
Abdullah ibn Umar said, when nicking the hump of his sacrificial animal to brand it (Asha'ar), "In the name of Allah, and Allah is greater."
Muwatta Imam Malik (link):
Nafi from Abdullah ibn Umar that when he brought an animal to be sacrificed from Madina he would garland it and brand it at Dhu'l-Hulayfa, doing the garlanding before the branding, but doing both in the same place, while facing the qibla. He would garland the animal with two sandals and brand it on its left side.
Even if cutting the live animal for blood, and shoes necklace are the Sunnah of Prophet Muhammad, but it seems that these are so shameful acts that present day Muslims are not performing them even if they claim to love the Sunnah of their Prophet.
 
Please also note, Kuffar used to sacrifice the animals in name of their gods, and Muhammad used to eat it before the prophethood, same as he used to worship those pagan gods. 
Sahih Bukhari, Book of Sacrifice and Hunting (link):
Allah's Apostle said that he met Zaid bin 'Amr Nufail at a place near Baldah and this had happened before Allah's Apostle received the Divine Inspiration. Allah's Apostle presented a dish of meat (that had been offered to him by the pagans) to Zaid bin 'Amr, but Zaid refused to eat of it and then said , "I do not eat of what you slaughter on your stonealtars (Ansabs) nor do I eat except that on which Allah's Name has been mentioned on slaughtering."
What type of Allah is he that he could not even keep his most beloved prophet in the universe to worship false gods and to eat meat of animals who were sacrificed in their name. 

 

قربانی ایک توہم پرستانہ و جاہلی رسم (از سعد رضا)
ـــــــــــــــــــــــــــــــــــــــــــــــــــــــــ

قربانی کی رسم دورِ قدیم سے ہی اس جاہلی اور توہم پرستی پر مبنی عقیدے کا جزو رہی ہے کہ خون بہانے سے خدا اور دیوتا راضی ہوتے ہیں۔ پہلے اس قربانی کی رسم میں انسانوں تک کو جانوروں کے ساتھ ساتھ خدا اور دیوتاؤں کی بھینٹ چڑھایا جاتا تھا لیکن پھر آہستہ آہستہ جب انسان تہذیب یافتہ ہونے لگا تو خدا اور دیوتا بھی محض جانوروں کے خون بہانے پر خوش ہونا سیکھ گئے۔

عرب میں بھی جانوروں کی قربانی کی یہ جاہلانہ رسم مکمل عروج پر تھی۔ خدا کے ساتھ ساتھ الگ الگ بتوں کے لئے بھی جانور قربان کئے جاتے تھے تاکہ خدا اور بت ان قربانیوں سے راضی ہو کر لوگوں کو برکت و خوشحالی سے نوازیں۔

محمد صلعم نے خود بھی اسی ماحول میں آنکھیں کھولی تھیں اور اپنے آباؤ اجداد کو قربانی کی انہی رسموں میں ملوث پایا تھا۔ دعویٰ نبوت سے بہت پہلے خود محمد صلعم بھی خدا کے ساتھ ساتھ بتوں کے نام پر دی جانے والی قربانیوں میں شریک رہے۔

چنانچہ حضرت عبداللہ ابن عمر رضی اللہ عنہ سے روایت ہے، وہ رسول اللہ صلی اللہ علیہ وسلم سے بیان کرتے ہیں کہ آپ نے زید بن عمرو بن نفیل سے مقام بلدح کے نشیبی علاقے میں ملاقات کی۔ یہ رسول اللہ صلی اللہ علیہ وسلم پر وحی نازل ہونے سے پہلے کا واقعہ ہے۔
رسول اللہ صلی اللہ علیہ وسلم نے اس کے آگے دسترخوان رکھا جس پر گوشت تھا۔ زید نے وہ گوشت کھانے سے انکار کر دیا، پھر کہا: "مَیں وہ گوشت نہیں کھاتا، جسے تم اپنے بتوں کے نام پر ذبح کرتے ہو، میں تو صرف وہی گوشت کھاتا ہوں جس پر (ذبح کرتے وقت) اللہ کا نام لیا گیا ہو۔"
(صحیح بخاری، کتاب الذبائح و الصید، باب ما ذبح علی النصب و الاصنام، حدیث 4599، مسند احمد، حدیث 5369)

یہ حدیث واضح ثبوت ہے کہ خود محمد صلعم تو بتوں کے نام پر دی جانے والی قربانیوں میں بھی شامل رہے۔ یہ الگ بات ہے کہ دعویٰ نبوت کے بعد غیر اللہ کے نام پر قربانی و ذبیحہ کرنے والے پر اللہ کی لعنت بھیجی (صحیح مسلم: حدیث 5124) لیکن اسی دور میں زید بن عمرو جیسے لوگ پہلے سے موجود تھے جو بتوں کے نام پر کی جانے والی قربانیوں سے سخت پرہیز کرتے تھے اور صرف اللہ کی ذات کو اس رسم کا حقدار سمجھتے تھے۔

محمد صلعم نے بھی آہستہ آہستہ بتوں کے نام پر کی جانے والی اس رسم کو صرف اللہ کی خاطر ہونے کا عقیدہ و نظریہ اپنا لیا۔ گویا وہی جاہلانہ و توہم پرستانہ رسم جو خدا کے ساتھ ساتھ بتوں کے نام پر بھی عرب میں رائج تھی، اسے ایک دین کے طور پر محض اللہ کے نام پر رائج کر دیا کہ بت تو نہیں لیکن اللہ ضرور جانوروں کا خون بہانے سے خوش ہوتا ہے۔ اسلامی قربانی کا مقصد و فلسفہ بھی محض خون بہا کر اللہ کو راضی کرنا ہی ہے اس کے لئے تفصیلی پوسٹ یہاں ملاحظہ کی جا سکتی ہے:

یہاں پر محمد صلعم کی قربانی سے متعلق چند اہم مسنون اعمال کا تذکرہ بھی ضروری سمجھتا ہوں جنہیں خود مسلمان آج بھلائے بیٹھے ہیں اور جو مزید اس بات کو مضبوط اور مئوکد کرتے ہیں کہ محمد صلعم کی اللہ کے نام پر قربانیاں بھی محض جاہلانہ و توہم پرستی پر مبنی رسومات کا مجموعہ تھیں۔

قربانی کے ان مسنون اعمال میں سے ہے کہ احرام کی حالت میں قربانی کے اونٹ / اونٹنی کو کوہان پر چیر لگا کر زخمی کر دیا جائے جسے اشعار کہا جاتا ہے اور پھر اس کے گلے میں جوتوں کا ہار ڈال دیا جائے، جسے قلادہ کہا جاتا ہے۔ دیکھئے صحیح مسلم (کتاب الحج، باب اشعار البدن و تقلیدہ عند الاحرام، حدیث 3016, آنلائن لنک)

محمد صلعم کی بیویاں خود اپنے ہاتھوں سے یہ قلادے (جوتوں کے ہار) تیار کرتی تھیں۔ (صحیح بخاری: حدیث 1698) محمد صلعم کے دیگر ساتھی بھی اپنی قربانیوں کو یہ ہار پہناتے اور قربانی کے اونٹ کی کوہان کی دائیں جانب چھری سے چیر کر زخم دے کر اشعار کی رسم سر انجام دیتے۔ (صحیح بخاری: حدیث قبل 1694)

جوتوں کا ہار اور قلادہ ڈالنے کی یہ توہم پرستانہ جاہلی رسم محض اونٹوں کے ساتھ مخصوص نہیں تھی بلکہ محمد صلعم نے قربانی کی بکریوں کو بھی یہ ہار پہنائے ہیں۔ (صحیح بخاری: حدیث 1702)

کیا کوئی بھی سلیم الفہم انسان ان حرکتوں کو معقول و اللہ کی رضا کا باعث سمجھ سکتا ہے؟ یقینی طور پر ہر انصاف پسند ان اعمال کو محض جاہلیت اور توہم پرستی کا شاخسانہ ہی سمجھے گا، اگر محض عقیدت انسان کو سوچنے سمجھنے سے بالکل بے بہرہ نہ کر دے۔
